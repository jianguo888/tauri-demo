## 三、Rust 后端实现

```rust
/// 星座运势 — POST /api/star
#[tauri::command]
async fn constellation(token: String, constellation: String) -> Result<String, String> {
    let client = reqwest::Client::new();
    let mut params = HashMap::new();
    params.insert("token", token.as_str());
    params.insert("constellation", constellation.as_str());
    let resp = client
        .post("https://v3.alapi.cn/api/star")
        .json(&params)
        .send()
        .await
        .map_err(|e| format!("请求失败: {e}"))?;
    let text = resp.text().await.map_err(|e| format!("读取响应失败: {e}"))?;
    Ok(text)
}
```

## 四、Vue 前端实现

### 4.1 响应式数据

```typescript
// ==================== 星座运势 ====================
const CONSTELLATIONS = [
  "白羊座", "金牛座", "双子座", "巨蟹座",
  "狮子座", "处女座", "天秤座", "天蝎座",
  "射手座", "摩羯座", "水瓶座", "双鱼座",
];
const selectedConstellation = ref("白羊座");
const starData = ref<any>(null);
const starLoading = ref(false);
const starError = ref("");

async function fetchConstellation() {
  starLoading.value = true;
  starError.value = "";
  starData.value = null;
  try {
    const res = await invoke<string>("constellation", {
      token: token.value,
      constellation: selectedConstellation.value,
    });
    const parsed = JSON.parse(res);
    if (parsed.success && parsed.data) {
      starData.value = parsed.data;
    } else {
      starError.value = parsed.message || "获取失败";
    }
  } catch (e) {
    starError.value = `请求出错: ${e}`;
  } finally {
    starLoading.value = false;
  }
}
```

### 4.2 模板

```html
<section class="card feature-card">
```
